  What is the top Six revenue movies?
In [27]:
top_revenues = pd.DataFrame(movies_df['revenue'].sort_values(ascending = False))
top_revenues['original_title'] = movies_df['original_title']
data = list(map(str,(top_revenues['original_title'])))
x = list(data[:6])
y = list(top_revenues['revenue'][:6])

# make a plot using pointplot for top 6 revenue movies.
ax = sns.pointplot(x=y,y=x)

#setup the figure size

#setup the title and labels of the plot.
